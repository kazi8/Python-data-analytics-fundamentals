#Assign grades
#Function to assign grades
def assign_grade(avg):
    if avg >= 90:
        return 'A'
    elif avg >= 80:
        return 'B'
    elif avg >= 70:
        return 'C'
    elif avg >= 60:
        return 'D'
    else:
        return 'F'

#Apply the function to calculate grades
df['Grade'] = df['Average'].apply(assign_grade)
